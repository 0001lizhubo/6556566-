CREATE TABLE course(ID VARCHAR(10),name VARCHAR(15) NOT NULL,cno VARCHAR(15) NOT NULL,cname  VARCHAR(20) NOT NULL,times VARCHAR(5),credit VARCHAR(5),kechengxingzhi VARCHAR(100),PRIMARY KEY(cno)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE  teacher(ID VARCHAR(10),name VARCHAR(15) NOT NULL,age VARCHAR(3),sex VARCHAR(10),education VARCHAR(100),department VARCHAR(100),school VARCHAR(200),health VARCHAR(40),title VARCHAR(40),post VARCHAR(40),PRIMARY KEY(ID)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;;

CREATE TABLE keyan(ID VARCHAR(10),name VARCHAR(15) NOT  NULL,direction VARCHAR(100),station VARCHAR(100),papers VARCHAR(100),PRIMARY KEY(ID)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;;

CREATE TABLE user(ID VARCHAR(10),password VARCHAR(100),role VARCHAR (10),PRIMARY KEY(ID)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;;
